    <script>
      let students = [];
      let currentIndex = 0;
      let currentTeacher = "";
      let currentSubject = "";
      let availableClasses = []; 

      // --- PASTE THIS MISSING FUNCTION ---
      function showLoading(isShowing, text = "") {
        const overlay = document.getElementById("loadingOverlay");
        const subtext = document.getElementById("loadingSubtext");
        if (isShowing) {
          overlay.style.display = "flex";
          subtext.innerText = text;
        } else {
          overlay.style.display = "none";
        }
      }
      // -----------------------------------

window.onload = function() {
  showLoading(true, "Connecting...");

  const studentCard = document.getElementById('studentCard');

  if (studentCard) {
    studentCard.addEventListener('touchstart', e => {
      touchStartX = e.changedTouches[0].screenX;
    }, {passive: true});

    studentCard.addEventListener('touchend', e => {
      touchEndX = e.changedTouches[0].screenX;
      if (touchStartX - touchEndX > 50) move(1);
      else if (touchEndX - touchStartX > 50) move(-1);
    }, {passive: true});
  }

  // ❌ REMOVE THIS LINE
  // document.getElementById("loginScreen").style.display = "block";

  // ✅ KEEP ONLY THIS
  document.getElementById("setupScreen").style.display = "none";
  document.getElementById("appScreen").style.display = "none";

  showLoading(false);
};



      function populateDropdown(data) {
        availableClasses = data;
        const dropdown = document.getElementById("classDropdown");
        const btn = document.getElementById("startBtn");
        
        dropdown.innerHTML = '<option value="" disabled selected>Select Teacher...</option>';
        if (data.length === 0) {
          dropdown.innerHTML = '<option value="" disabled selected>No data found!</option>';
          showLoading(false);
          return;
        }

        data.forEach((cls, index) => {
          let option = document.createElement("option");
          option.value = index; 
          option.text = `${cls.teacher} - ${cls.subject}`;
          dropdown.appendChild(option);
        });

        btn.disabled = false; 
        showLoading(false);
      }

      function startApp() {
        const dropValue = document.getElementById("classDropdown").value;
        if (dropValue === "") return;

        const selectedClass = availableClasses[dropValue];
        currentTeacher = selectedClass.teacher;
        currentSubject = selectedClass.subject;

        document.getElementById("setupScreen").style.display = "none";
        document.getElementById("appScreen").style.display = "block";
        
        const dateInput = document.getElementById('dateSelect');
        if (!dateInput.value) { dateInput.value = new Date().toISOString().split('T')[0]; }
        loadData();
      }

      function loadData() {
        showLoading(true, "Fetching students...");
        const selectedDate = document.getElementById('dateSelect').value;
        
        google.script.run
          .withSuccessHandler(data => {
            students = data;
            showLoading(false);
            currentIndex = 0; 
            if(students.length === 0) {
              alert("No students found!");
            } else {
              buildJumpMenu();
              updateUI();
            }
          })
          .withFailureHandler(error => { showLoading(false); alert("Error: " + error.message); })
          .getStudents(selectedDate, currentTeacher, currentSubject);
      }

      function buildJumpMenu() {
        const drop = document.getElementById("nameDisplay");
        drop.innerHTML = "";
        students.forEach((s, i) => {
          let opt = document.createElement("option");
          opt.value = i;
          opt.text = (i + 1) + ". " + s.name;
          drop.appendChild(opt);
        });
      }

      function jumpToStudent() {
        const drop = document.getElementById("nameDisplay");
        currentIndex = parseInt(drop.value);
        updateUI();
      }

      function updateUI() {
        const card = document.getElementById('studentCard');
        card.className = ''; 

        if (students.length > 0) {
          const s = students[currentIndex];
          document.getElementById('nameDisplay').value = currentIndex;
          document.getElementById('rollBadge').innerText = "ROLL #" + s.roll;
          
          const statusDiv = document.getElementById('statusDisplay');
          statusDiv.innerText = "Status: " + s.status;
          statusDiv.className = "status-label status-" + s.status.replace(" ", ".");
          
          document.getElementById('counter').innerText = `STUDENT ${currentIndex + 1} OF ${students.length}`;
        }
      }

      function log(status) {
        if (students.length === 0) return;
        const s = students[currentIndex];
        const selectedDate = document.getElementById('dateSelect').value;
        s.status = status;
        
        const card = document.getElementById('studentCard');
        card.classList.add('fade-out');

        google.script.run.recordAttendance(s.name, s.enrollment, status, selectedDate, currentTeacher, currentSubject);

        setTimeout(() => {
          if (currentIndex < students.length - 1) {
            currentIndex++;
            card.classList.remove('fade-out');
            card.classList.add('fade-in');
            updateUI();
            setTimeout(() => card.classList.remove('fade-in'), 100);
          } else {
            document.getElementById('studentCard').innerHTML =
  "<h2 style='margin: 40px 0;'>Done! 🎉</h2>" +
  "<button onclick='sendWhatsAppMessages()' style='width:100%; background:#25D366; color:white; padding:15px; border-radius:12px; border:none; margin-bottom:10px;'>Send WhatsApp Alerts</button>" +
  "<button onclick='goToSetup()' style='width:100%; background:#1a73e8; color:white; padding:15px; border-radius:12px; border:none;'>Back to Setup</button>";
            document.getElementById('counter').innerText = "";
          }
        }, 200);
      }

      function move(step) {
        if (students.length === 0) return;
        currentIndex += step;
        if (currentIndex < 0) currentIndex = 0;
        if (currentIndex >= students.length) currentIndex = students.length - 1;
        updateUI();
      }

      let touchEndX = 0;

  document.getElementById("adminScreen").style.display = "none";

}

// SAVE TEACHER
function saveTeacher() {
  let name = document.getElementById("tName").value;
  let subject = document.getElementById("tSubject").value;

  if (!name || !subject) {
    alert("Fill all fields");
    return;
  }

  google.script.run
    .withSuccessHandler(msg => {
      alert(msg);
      closeAdmin();
      google.script.run
      .withSuccessHandler(populateDropdown)
      .getTeacherList();
    })
    .addTeacher(name, subject);
}

// SAVE STUDENT
function saveStudent() {
  let name = document.getElementById("sName").value.trim();
  let enroll = document.getElementById("sEnroll").value.trim();
  let roll = document.getElementById("sRoll").value.trim();
  let subject = document.getElementById("sSubject").value.trim();
  let phone = document.getElementById("sPhone").value.trim();
// ✅ Validation
if (!name || !enroll || !roll || !subject || !phone) {
  alert("⚠️ Please fill all fields");
  return;
}
// Optional: Roll must be number
  if (isNaN(roll)) {
    alert("⚠️ Roll number must be numeric");
    return;
  }
  // ✅ Phone validation
if (isNaN(phone)) {
  alert("⚠️ Phone number must contain only digits");
  return;
}

// ✅ Length check (India = 10 digits)
if (phone.length !== 10) {
  alert("⚠️ Phone number must be 10 digits");
  return;
}

// ✅ Optional: Starts with valid digits (6–9 in India)
if (!/^[6-9]/.test(phone)) {
  alert("⚠️ Invalid Indian phone number");
  return;
}
 // ✅ Show loading
  showLoading(true, "Saving student...");

  google.script.run
    .withSuccessHandler(msg => {
      showLoading(false);
      alert("✅ " + msg);

      // ✅ Clear form after success
      document.getElementById("sName").value = "";
      document.getElementById("sEnroll").value = "";
      document.getElementById("sRoll").value = "";
      document.getElementById("sSubject").value = "";
      document.getElementById("sPhone").value = "";
    })
    .withFailureHandler(err => {
      showLoading(false);
      alert("❌ Error: " + err.message);
    })
    .addStudent(name, enroll, roll, subject,phone);
}
function loadSubjectDropdown() {
  const dropdown = document.getElementById("sSubject");

  // Clear existing
  dropdown.innerHTML = '<option value="" disabled selected>Select Subject</option>';

  // Use already loaded data
  let subjects = new Set();

  availableClasses.forEach(cls => {
    subjects.add(cls.subject);
  });

  subjects.forEach(sub => {
    let opt = document.createElement("option");
    opt.value = sub;
    opt.text = sub;
    dropdown.appendChild(opt);
  });
}
function goToSetup() {
  // Switch screens
  document.getElementById("appScreen").style.display = "none";
  document.getElementById("adminScreen").style.display = "none";
  document.getElementById("setupScreen").style.display = "block";

  // Reset student card UI
  document.getElementById("studentCard").innerHTML = `
    <div id="rollBadge" class="roll-badge">Roll #--</div>
    <select id="nameDisplay" class="name-display" onchange="jumpToStudent()"></select>
    <div id="statusDisplay" class="status-label status-Not.Marked">Status: Not Marked</div>
    <div class="btn-group">
      <button class="attendance-btn present" onclick="log('Present')">PRESENT</button>
      <button class="attendance-btn absent" onclick="log('Absent')">ABSENT</button>
    </div>
  `;

  // Reset variables
  students = [];
  currentIndex = 0;

  // ✅ THIS LINE GOES HERE
  //google.script.run.withSuccessHandler(populateDropdown).getTeacherList();
  google.script.run
    .withSuccessHandler(populateDropdown)
    .withFailureHandler(err => alert("Error: " + err.message))
    .getTeacherList();
}

function sendWhatsAppMessages() {
  const selectedDate = document.getElementById('dateSelect').value;

  showLoading(true, "Fetching absent students...");

  google.script.run
  .withSuccessHandler(data => {
    showLoading(false);

    if (data.length === 0) {
      alert("No absent students 🎉");
      return;
    }

    let html = `
      <h3>Absent Students</h3>
      <button onclick="goToSetup()" 
        style="width:100%; background:#1a73e8; color:white; padding:12px; border-radius:10px; border:none; margin-bottom:15px;">
        ⬅ Back to Setup
      </button>
    `;

    data.forEach(st => {
      html += `
        <div id="row_${st.row}" style="margin:10px; padding:10px; border:1px solid #ccc;">
          <b>${st.name}</b><br>
          <button onclick="sendSingle('${st.link}', ${st.row})">
            Send WhatsApp
          </button>
        </div>
      `;
    });

    html += `
      <br>
      <button onclick="goToSetup()" 
        style="width:100%; background:#1a73e8; color:white; padding:12px; border-radius:10px; border:none;">
        ⬅ Back to Setup
      </button>
    `;

    document.getElementById('studentCard').innerHTML = html;
  })
  .withFailureHandler(error => {
    showLoading(false);
    alert("Backend Error: " + error.message);
  })
  .getAbsentForWhatsApp(selectedDate, currentTeacher, currentSubject);
}

function sendSingle(link, row) {
  const selectedDate = document.getElementById('dateSelect').value;

  // Open WhatsApp
if (window.AppInventor) {
  AppInventor.setWebViewString(link);
} else {
  window.open(link, "_blank");
}
  // Remove from UI
  const div = document.getElementById("row_" + row);
  if (div) div.remove();

  // Mark as sent in sheet
  google.script.run.markWhatsAppSent(row, selectedDate, currentTeacher, currentSubject);
}
function login() {
  let u = document.getElementById("user").value;
  let p = document.getElementById("pass").value;

  google.script.run
    .withSuccessHandler(function(res) {

      if (res !== "fail") {

        document.getElementById("loginScreen").style.display = "none";
        document.getElementById("setupScreen").style.display = "flex";

        google.script.run
          .withSuccessHandler(populateDropdown)
          .getTeacherList();

      } else {
        alert("Invalid credentials ❌");
      }
    })
    .checkLogin(u, p);
}
